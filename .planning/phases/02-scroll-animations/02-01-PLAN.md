# GSAP ScrollTrigger Pattern (from research)
```javascript
import { gsap } from "gsap";
import { ScrollTrigger } from "gsap/ScrollTrigger";

gsap.registerPlugin(ScrollTrigger);

// Batch reveal for multiple elements
ScrollTrigger.batch(".reveal-element", {
  onEnter: (elements) => {
    gsap.from(elements, {
      opacity: 0,
      y: 50,
      stagger: 0.15,
      duration: 0.8,
      ease: "power2.out"
    });
  },
  start: "top 80%", // Trigger when element is 80% from top
});
```

```jsx
// In useEffect, after component mount
useEffect(() => {
  gsap.registerPlugin(ScrollTrigger);
  
  // Reveal animations for sections
  ScrollTrigger.batch(".section-reveal", {
    onEnter: (elements) => {
      gsap.from(elements, {
        opacity: 0,
        y: 60,
        stagger: 0.12,
        duration: 0.7,
        ease: "power2.out"
      });
    },
    start: "top 85%",
  });
  
  return () => ScrollTrigger.getAll().forEach(t => t.kill());
}, []);
```
